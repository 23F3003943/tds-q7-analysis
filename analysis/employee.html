Q7 - Employee Department Analysis

This HTML file contains the Python code used for analysis.

import pandas as pd
import matplotlib.pyplot as plt

# Create sample dataset
data = {
    "employee_id": list(range(1,101)),
    "department": ["Sales","IT","HR","Finance","Marketing"] * 20,
}

df = pd.DataFrame(data)

# Frequency of Sales department
count_sales = df[df["department"]=="Sales"].shape[0]
print("Sales Department Count:", count_sales)

# Histogram
plt.figure(figsize=(6,6))
df["department"].value_counts().plot(kind="bar")
plt.title("Department Distribution")
plt.xlabel("Department")
plt.ylabel("Count")
plt.savefig("hist.png")
